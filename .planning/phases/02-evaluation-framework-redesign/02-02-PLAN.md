       ```python
       """Leakage prevention for COT feedback.

       Per D-06: COT feedback must only use training-set analysis.
       No validation or test set information should leak into the LLM prompt.
       """

       import re
       from typing import Dict, List

       # Metrics that are safe to include in COT feedback (from training set only)
       COT_SAFE_KEYS = {'sharpe', 'max_dd', 'total_return'}

       # Patterns that indicate validation/test data leaking into prompts
       LEAKAGE_PATTERNS = [
           r'\bval(?:idation)?[_\s]?(?:sharpe|max_dd|return|sortino|calmar|win_rate|drawdown)',
           r'\btest[_\s]?(?:sharpe|max_dd|return|sortino|calmar|win_rate|drawdown)',
           r'\bvalidation[_\s]set\b',
           r'\btest[_\s]set\b',
           r'\bout[_\s]of[_\s]sample\b',
       ]

       def filter_cot_metrics(metrics: Dict) -> Dict:
           """Strip non-training metrics from results before passing to COT feedback.

           Only allow sharpe, max_dd, total_return to pass through.
           Remove sortino, calmar, win_rate, daily_returns, regime_vectors,
           trades, regime_metrics -- these could encode validation/test information.
           """
           return {k: v for k, v in metrics.items() if k in COT_SAFE_KEYS}

       def check_prompt_for_leakage(prompt_text: str) -> List[str]:
           """Check prompt text for signs of validation/test data leakage.

           Returns list of matched strings. Empty list means no leak detected.
           """
           matches = []
           for pattern in LEAKAGE_PATTERNS:
               found = re.findall(pattern, prompt_text, re.IGNORECASE)
               matches.extend(found)
           return matches
       ```

       ```python
       """Regime-stratified evaluation.

       Per D-07: Use existing regime_detector.py (3-dim: trend/volatility/risk)
       to label test periods, then report per-regime Sharpe and MaxDD.

       Regime buckets based on trend_direction (index 0 of regime vector):
         bull: trend > 0.3
         bear: trend < -0.3
         sideways: -0.3 <= trend <= 0.3
       """

       import numpy as np
       from typing import Dict, List
       from evaluation.metrics import sharpe_ratio, max_drawdown, sortino

       MIN_REGIME_DAYS = 10  # Flag regimes with fewer days as low confidence

       def classify_regime(regime_vector) -> str:
           """Classify a regime vector into bull/bear/sideways.

           Uses trend_direction (index 0) with thresholds +/- 0.3.
           Per established convention from regime_detector.py.

           Args:
               regime_vector: 3-element array-like [trend, volatility, risk]

           Returns:
               "bull", "bear", or "sideways"
           """
           trend = float(regime_vector[0])
           if trend > 0.3:
               return "bull"
           elif trend < -0.3:
               return "bear"
           else:
               return "sideways"

       def evaluate_by_regime(
           daily_returns: List[float],
           regime_vectors: List,
           min_days: int = MIN_REGIME_DAYS
       ) -> Dict[str, Dict]:
           """Compute per-regime metrics from evaluation loop outputs.

           Per D-07: Report per-regime Sharpe and MaxDD.
           Per Pitfall 1 from research: Flag regimes with < 60 days as low confidence.

           Args:
               daily_returns: List of daily return floats
               regime_vectors: List of 3-element regime vectors (one per return)
               min_days: Minimum days to not flag as low confidence

           Returns:
               Dict with keys "bull", "bear", "sideways", each containing:
                 sharpe, max_dd, count, low_confidence
           """
           if len(daily_returns) == 0:
               return {
                   "bull": {"sharpe": 0.0, "max_dd": 0.0, "count": 0, "low_confidence": True},
                   "bear": {"sharpe": 0.0, "max_dd": 0.0, "count": 0, "low_confidence": True},
                   "sideways": {"sharpe": 0.0, "max_dd": 0.0, "count": 0, "low_confidence": True},
               }

           # Bucket returns by regime
           buckets = {"bull": [], "bear": [], "sideways": []}
           for ret, rv in zip(daily_returns, regime_vectors):
               regime_name = classify_regime(rv)
               buckets[regime_name].append(ret)

           results = {}
           for name, returns in buckets.items():
               count = len(returns)
               if count >= 2:
                   s = sharpe_ratio(returns)
                   dd = max_drawdown(returns)
               else:
                   s = 0.0
                   dd = 0.0
               results[name] = {
                   "sharpe": s,
                   "max_dd": dd,
                   "count": count,
                   "low_confidence": count < min_days,
               }
           return results
       ```
