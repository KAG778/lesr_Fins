       ```python
       """Cross-experiment aggregation and reporting.

       Per D-08: Build on Phase 1's analyze_existing.py to aggregate results
       across result_SW* directories into publication-ready comparison tables.
       Per EVAL-05: Cross-stock, cross-window, cross-run comparison reports.
       """

       import pickle
       from pathlib import Path
       from typing import Optional

       import numpy as np
       import pandas as pd

       DEFAULT_STOCKS = ['TSLA', 'NFLX', 'AMZN', 'MSFT']


       def aggregate_cross_experiment(
           base_dir: str,
           pattern: str = "result_SW*_test*",
           stocks: list = None
       ) -> pd.DataFrame:
           """Aggregate results from all sliding window experiments into a DataFrame.

           Args:
               base_dir: Root directory containing result_SW* directories
               pattern: Glob pattern for finding result directories
               stocks: List of stock tickers to include (default: TSLA, NFLX, AMZN, MSFT)

           Returns:
               DataFrame with columns: window, test_year, ticker, method,
               sharpe, max_dd, total_return, sortino, calmar, win_rate
           """
           if stocks is None:
               stocks = DEFAULT_STOCKS

           columns = ['window', 'test_year', 'ticker', 'method',
                       'sharpe', 'max_dd', 'total_return',
                       'sortino', 'calmar', 'win_rate']

           all_rows = []
           for result_dir in sorted(Path(base_dir).glob(pattern)):
               pkl_file = result_dir / 'test_set_results.pkl'
               if not pkl_file.exists():
                   continue

               # Parse directory name: result_SW01_test2015
               name = result_dir.name
               try:
                   sw_part = name.split('SW')[1].split('_')[0]
                   window = int(sw_part)
                   test_year = int(name.split('test')[1])
               except (IndexError, ValueError):
                   continue

               try:
                   with open(pkl_file, 'rb') as f:
                       data = pickle.load(f)
               except Exception:
                   continue

               for ticker in stocks:
                   if ticker not in data or data[ticker].get('error'):
                       continue
                   for method_key, method_label in [('lesr_test', 'LESR'), ('baseline_test', 'Baseline')]:
                       metrics = data[ticker].get(method_key, {})
                       if not metrics:
                           continue
                       all_rows.append({
                           'window': window,
                           'test_year': test_year,
                           'ticker': ticker,
                           'method': method_label,
                           'sharpe': metrics.get('sharpe', 0.0),
                           'max_dd': metrics.get('max_dd', 0.0),
                           'total_return': metrics.get('total_return', 0.0),
                           'sortino': metrics.get('sortino', 0.0),
                           'calmar': metrics.get('calmar', 0.0),
                           'win_rate': metrics.get('win_rate', 0.0),
                       })

           if all_rows:
               return pd.DataFrame(all_rows, columns=columns)
           return pd.DataFrame(columns=columns)


       def generate_report(
           df: pd.DataFrame,
           format: str = "markdown",
           group_by: str = "ticker"
       ) -> str:
           """Generate publication-ready comparison table.

           Args:
               df: DataFrame from aggregate_cross_experiment()
               format: "markdown" or "latex"
               group_by: How to group results ("ticker", "window", or "method")

           Returns:
               Formatted table string
           """
           if df.empty:
               return "No data available for report."

           # Summary statistics
           summary = df.groupby('method').agg({
               'sharpe': 'mean',
               'max_dd': 'mean',
               'total_return': 'mean',
               'sortino': 'mean',
               'calmar': 'mean',
               'win_rate': 'mean',
           }).round(3)

           if format == "latex":
               # LaTeX table
               lines = [
                   r"\begin{table}[htbp]",
                   r"\centering",
                   r"\caption{Cross-Experiment Performance Comparison}",
                   r"\begin{tabular}{lrrrrrr}",
                   r"\hline",
                   "Method & Sharpe & MaxDD (\%) & Return (\%) & Sortino & Calmar & Win Rate \\\\",
                   r"\hline",
               ]
               for method, row in summary.iterrows():
                   lines.append(
                       f"{method} & {row['sharpe']:.3f} & {row['max_dd']:.2f} & "
                       f"{row['total_return']:.2f} & {row['sortino']:.3f} & "
                       f"{row['calmar']:.3f} & {row['win_rate']:.3f} \\\\"
                   )
               lines.extend([
                   r"\hline",
                   r"\end{tabular}",
                   r"\end{table}",
               ])
               return '\n'.join(lines)

           else:
               # Markdown table
               lines = [
                   "# Cross-Experiment Performance Comparison\n",
                   "## Summary Statistics\n",
                   "| Method | Sharpe | MaxDD (%) | Return (%) | Sortino | Calmar | Win Rate |",
                   "|--------|--------|-----------|------------|---------|--------|----------|",
               ]
               for method, row in summary.iterrows():
                   lines.append(
                       f"| {method} | {row['sharpe']:.3f} | {row['max_dd']:.2f} | "
                       f"{row['total_return']:.2f} | {row['sortino']:.3f} | "
                       f"{row['calmar']:.3f} | {row['win_rate']:.3f} |"
                   )

               # Per-group detail
               if group_by == "ticker":
                   lines.append("\n## Per-Ticker Detail\n")
                   for ticker in sorted(df['ticker'].unique()):
                       t_df = df[df['ticker'] == ticker]
                       lines.append(f"\n### {ticker}\n")
                       sub = t_df.groupby('method').agg({
                           'sharpe': 'mean', 'max_dd': 'mean', 'total_return': 'mean',
                           'sortino': 'mean', 'calmar': 'mean', 'win_rate': 'mean',
                       }).round(3)
                       lines.append(sub.to_markdown())

               return '\n'.join(lines)
       ```
