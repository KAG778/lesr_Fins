```python
def _sharpe(self, returns, rf=0.0):
    if len(returns) < 2: return 0.0
    r = np.array(returns)
    m, s = r.mean() * 252, r.std() * np.sqrt(252)
    return (m - rf) / s if s > 0 else 0.0

def _max_dd(self, returns):
    if len(returns) < 2: return 0.0
    cum = np.cumsum(returns)
    peak = np.maximum.accumulate(cum)
    return abs((cum - peak).min()) * 100
```

       ```python
       import numpy as np

       def sharpe_ratio(returns, rf=0.0):
           """Annualized Sharpe ratio. rf is annualized risk-free rate."""
           if len(returns) < 2:
               return 0.0
           r = np.array(returns, dtype=float)
           annualized_mean = r.mean() * 252
           annualized_std = r.std() * np.sqrt(252)
           if annualized_std < 1e-10:
               return 0.0
           return float((annualized_mean - rf) / annualized_std)

       def max_drawdown(returns):
           """Maximum drawdown as positive percentage (e.g., 15.3 means 15.3%)."""
           if len(returns) < 2:
               return 0.0
           cum = np.cumsum(np.array(returns, dtype=float))
           peak = np.maximum.accumulate(cum)
           dd = cum - peak
           return float(abs(dd.min())) * 100

       def sortino(returns, rf=0.0):
           """Annualized Sortino ratio using downside deviation only."""
           if len(returns) < 2:
               return 0.0
           r = np.array(returns, dtype=float)
           annualized_mean = r.mean() * 252 - rf
           downside = r[r < 0]
           if len(downside) == 0:
               return float('inf') if annualized_mean > 0 else 0.0
           downside_std = downside.std() * np.sqrt(252)
           if downside_std < 1e-10:
               return 0.0
           return float(annualized_mean / downside_std)

       def calmar(returns, rf=0.0):
           """Calmar ratio: annualized return / max drawdown."""
           if len(returns) < 2:
               return 0.0
           annual_return = np.mean(returns) * 252 - rf
           mdd = max_drawdown(returns)
           if mdd < 1e-8:
               return float('inf') if annual_return > 0 else 0.0
           return float(annual_return / (mdd / 100))

       def win_rate(returns):
           """Fraction of positive return periods."""
           if len(returns) < 1:
               return 0.0
           r = np.array(returns, dtype=float)
           positive = (r > 0).sum()
           return float(positive / len(r))
       ```
