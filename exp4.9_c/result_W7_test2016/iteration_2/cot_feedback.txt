import numpy as np

def revise_state(s):
    # s: 120d raw state
    n_days = 20
    closing_prices = s[0::6][:n_days]  # Extract last 20 closing prices
    volumes = s[4::6][:n_days]  # Extract last 20 trading volumes

    # Feature 1: Price Momentum (current closing price - moving average of last 10 days)
    if n_days > 10:
        moving_average_10 = np.mean(closing_prices[-10:])
        momentum = closing_prices[-1] - moving_average_10
    else:
        momentum = 0

    # Feature 2: Average Trading Volume over the last 10 days
    avg_volume = np.mean(volumes[-10:]) if len(volumes) >= 10 else np.mean(volumes)

    # Feature 3: Historical Volatility (Standard Deviation of returns over the last 20 days)
    returns = np.diff(closing_prices) / closing_prices[:-1]
    historical_volatility = np.std(returns) if len(returns) >= 20 else 0

    # Feature 4: Average True Range (ATR) to capture market volatility
    high_prices = s[2::6][:n_days]  # Extract high prices
    low_prices = s[3::6][:n_days]   # Extract low prices
    atr = np.mean(np.maximum(high_prices[1:] - low_prices[1:], 
                              np.abs(high_prices[1:] - closing_prices[:-1]),
                              np.abs(low_prices[1:] - closing_prices[:-1]))) if len(high_prices) > 1 else 0

    # Feature 5: 14-day Relative Strength Index (RSI)
    def compute_rsi(prices, period=14):
        if len(prices) < period:
            return 50  # Neutral RSI if not enough data
        deltas = np.diff(prices)
        gain = np.where(deltas > 0, deltas, 0).mean()
        loss = -np.where(deltas < 0, deltas, 0).mean()
        if (gain + loss) == 0:
            return 50  # Neutral RSI
        return 100 - (100 / (1 + gain / loss))
    
    rsi = compute_rsi(closing_prices)

    features = [momentum, avg_volume, historical_volatility, atr, rsi]
    return np.array(features)
