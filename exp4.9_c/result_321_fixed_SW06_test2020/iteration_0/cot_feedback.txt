    rsi_period = 14
    if len(closing_prices) >= rsi_period:
        delta = closing_prices[-rsi_period:] - closing_prices[-rsi_period-1:-1]

        rsi = 100 - (100 / (1 + rs))
    else:
        rsi = 50  # Neutral RSI if not enough data

    return np.array([price_change, avg_volume, rsi])

    price_change = np.zeros(20)
    for i in range(1, 20):
        if closing_prices[i - 1] != 0:  # Avoid division by zero
            price_change[i] = (closing_prices[i] - closing_prices[i - 1]) / closing_prices[i - 1]

            price_change[i] = 0.0

            volume_change[i] = 0.0
