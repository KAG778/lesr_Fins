```python
from dataclasses import dataclass
from typing import Tuple
import torch

@dataclass
class TrainingConfig:
    """训练配置基类"""
    # 环境相关
    env_id: str = "CartPole-v1"
    seed: int = 42
    
    # 训练相关
    total_timesteps: int = 1_000_000
    learning_rate: float = 3e-4
    batch_size: int = 64
    n_epochs: int = 10
    
    # 算法相关
    gamma: float = 0.99
    gae_lambda: float = 0.95
    
    # 设备
    device: str = "auto"
    
    # 日志
    log_interval: int = 1000
    save_interval: int = 10000
    eval_interval: int = 5000
    
    def __post_init__(self):
        if self.device == "auto":
            self.device = "cuda" if torch.cuda.is_available() else "cpu"

@dataclass
class PPOConfig(TrainingConfig):
    """PPO特定配置"""
    n_steps: int = 2048
    batch_size: int = 64
    n_epochs: int = 10
    clip_range: float = 0.2
    ent_coef: float = 0.01
    vf_coef: float = 0.5
    max_grad_norm: float = 0.5

@dataclass
class SACConfig(TrainingConfig):
    """SAC特定配置"""
    buffer_size: int = 1_000_000
    learning_starts: int = 10000
    train_freq: int = 1
    gradient_steps: int = 1
    ent_coef: str = "auto"  # 或指定具体值
    target_entropy: str = "auto"
    tau: float = 0.005
```

### 2.1 完整训练脚本

```python
# scripts/train.py
import argparse
import os
import yaml
from pathlib import Path
import torch
import numpy as np
from tqdm import tqdm
import wandb

from stable_baselines3 import PPO, SAC, TD3
from stable_baselines3.common.env_util import make_vec_env
from stable_baselines3.common.vec_env import SubprocVecEnv, VecNormalize
from stable_baselines3.common.callbacks import (
    EvalCallback,
    CheckpointCallback,
    WandBCallback,
    StopTrainingOnRewardThreshold
)

def setup_wandb(config, project_name="rl-experiments"):
    """初始化wandb"""
    wandb.init(
        project=project_name,
        config=config,
        name=config.get('experiment_name', 'unnamed')
    )

def create_environment(config):
    """创建训练环境"""
    # 创建向量化环境
    env = make_vec_env(
        config['env_id'],
        n_envs=config.get('n_envs', 1),
        seed=config['seed']
    )
    
    # 归一化观测和奖励
    env = VecNormalize(
        env,
        norm_obs=True,
        norm_reward=True,
        clip_obs=10.0,
        clip_reward=10.0
    )
    
    return env

def create_callbacks(config):
    """创建回调函数"""
    callbacks = []
    
    # 1. 评估回调
    eval_env = make_vec_env(config['env_id'], n_envs=1)
    eval_callback = EvalCallback(
        eval_env,
        best_model_save_path=f"./logs/{config['experiment_name']}/best_model",
        log_path=f"./logs/{config['experiment_name']}/eval_results",
        eval_freq=config.get('eval_interval', 5000),
        deterministic=True,
        render=False,
        n_eval_episodes=10
    )
    callbacks.append(eval_callback)
    
    # 2. 检查点回调
    checkpoint_callback = CheckpointCallback(
        save_freq=config.get('save_interval', 10000),
        save_path=f"./checkpoints/{config['experiment_name']}",
        name_prefix="rl_model"
    )
    callbacks.append(checkpoint_callback)
    
    # 3. WandB回调
    if config.get('use_wandb', False):
        wandb_callback = WandBCallback(
            model_save_path=f"./logs/{config['experiment_name']}/wandb",
            verbose=2
        )
        callbacks.append(wandb_callback)
    
    # 4. 停止训练回调（达到目标奖励）
    if config.get('target_reward'):
        stop_callback = StopTrainingOnRewardThreshold(
            reward_threshold=config['target_reward'],
            verbose=1
        )
        callbacks.append(stop_callback)
    
    return callbacks

def train(config):
    """训练函数"""
    # 设置随机种子
    torch.manual_seed(config['seed'])
    np.random.seed(config['seed'])
    
    # 创建环境
    train_env = create_environment(config)
    
    # 创建算法
    algorithm = config['algorithm'].lower()
    if algorithm == 'ppo':
        model = PPO(
            "MlpPolicy",
            train_env,
            learning_rate=config['learning_rate'],
            n_steps=config.get('n_steps', 2048),
            batch_size=config['batch_size'],
            n_epochs=config.get('n_epochs', 10),
            gamma=config['gamma'],
            gae_lambda=config.get('gae_lambda', 0.95),
            clip_range=config.get('clip_range', 0.2),
            ent_coef=config.get('ent_coef', 0.01),
            vf_coef=config.get('vf_coef', 0.5),
            max_grad_norm=config.get('max_grad_norm', 0.5),
            tensorboard_log=f"./logs/{config['experiment_name']}/tensorboard",
            verbose=1,
            device=config.get('device', 'auto')
        )
    elif algorithm == 'sac':
        model = SAC(
            "MlpPolicy",
            train_env,
            learning_rate=config['learning_rate'],
            buffer_size=config.get('buffer_size', 1_000_000),
            learning_starts=config.get('learning_starts', 10000),
            batch_size=config['batch_size'],
            tau=config.get('tau', 0.005),
            gamma=config['gamma'],
            train_freq=config.get('train_freq', 1),
            gradient_steps=config.get('gradient_steps', 1),
            ent_coef=config.get('ent_coef', 'auto'),
            tensorboard_log=f"./logs/{config['experiment_name']}/tensorboard",
            verbose=1,
            device=config.get('device', 'auto')
        )
    else:
        raise ValueError(f"Unknown algorithm: {algorithm}")
    
    # 创建回调
    callbacks = create_callbacks(config)
    
    # 训练
    print(f"Starting training with {algorithm.upper()}...")
    print(f"Total timesteps: {config['total_timesteps']}")
    
    model.learn(
        total_timesteps=config['total_timesteps'],
        callback=callbacks
    )
    
    # 保存最终模型
    model.save(f"./checkpoints/{config['experiment_name']}/final_model")
    train_env.save(f"./checkpoints/{config['experiment_name']}/vec_normalize.pkl")
    
    print("Training completed!")
    
    return model

def main():
    parser = argparse.ArgumentParser(description='Train RL agent')
    parser.add_argument('--config', type=str, required=True, help='Path to config file')
    args = parser.parse_args()
    
    # 加载配置
    with open(args.config, 'r') as f:
        config = yaml.safe_load(f)
    
    # 设置wandb
    if config.get('use_wandb', False):
        setup_wandb(config)
    
    # 训练
    model = train(config)

if __name__ == "__main__":
    main()
```

```python
# src/utils/monitoring.py
import time
import psutil
import torch

class TrainingMonitor:
    """训练监控器"""
    
    def __init__(self, log_dir):
        self.log_dir = Path(log_dir)
        self.log_dir.mkdir(parents=True, exist_ok=True)
        self.metrics = []
        self.start_time = time.time()
    
    def log_metrics(self, episode, rewards, losses, model):
        """记录训练指标"""
        # 基础指标
        metrics = {
            'episode': episode,
            'timestamp': time.time() - self.start_time,
            'mean_reward': np.mean(rewards),
            'std_reward': np.std(rewards),
            'min_reward': np.min(rewards),
            'max_reward': np.max(rewards),
            'episode_length': len(rewards),
        }
        
        # 损失指标
        if losses:
            metrics.update({
                'policy_loss': np.mean([l['policy_loss'] for l in losses]),
                'value_loss': np.mean([l['value_loss'] for l in losses]),
                'entropy': np.mean([l['entropy'] for l in losses]),
            })
        
        # 系统指标
        metrics.update({
            'cpu_percent': psutil.cpu_percent(),
            'memory_percent': psutil.virtual_memory().percent,
            'gpu_memory_used': torch.cuda.memory_allocated() / 1024**3 if torch.cuda.is_available() else 0,
        })
        
        # 网络统计
        if hasattr(model, 'policy'):
            metrics.update(self.get_network_stats(model.policy))
        
        self.metrics.append(metrics)
        
        # 打印进度
        if episode % 10 == 0:
            self.print_progress(metrics)
    
    def get_network_stats(self, network):
        """获取网络统计"""
        stats = {}
        
        for name, param in network.named_parameters():
            if param.grad is not None:
                stats[f'{name}_grad_norm'] = param.grad.norm().item()
                stats[f'{name}_weight_norm'] = param.data.norm().item()
        
        return stats
    
    def print_progress(self, metrics):
        """打印训练进度"""
        print(f"\nEpisode {metrics['episode']}")
        print(f"Mean Reward: {metrics['mean_reward']:.2f} ± {metrics['std_reward']:.2f}")
        print(f"Time: {metrics['timestamp']:.1f}s")
        
        if 'policy_loss' in metrics:
            print(f"Policy Loss: {metrics['policy_loss']:.4f}")
            print(f"Value Loss: {metrics['value_loss']:.4f}")
            print(f"Entropy: {metrics['entropy']:.4f}")
        
        print(f"CPU: {metrics['cpu_percent']:.1f}% | Memory: {metrics['memory_percent']:.1f}%")
        if metrics['gpu_memory_used'] > 0:
            print(f"GPU Memory: {metrics['gpu_memory_used']:.2f} GB")
    
    def save_metrics(self, filename='training_metrics.json'):
        """保存指标到文件"""
        import json
        with open(self.log_dir / filename, 'w') as f:
            json.dump(self.metrics, f, indent=2)
```

```python
# src/utils/visualization.py
import matplotlib.pyplot as plt
import numpy as np
from pathlib import Path

class RLVisualizer:
    """强化学习可视化工具"""
    
    def __init__(self, save_dir):
        self.save_dir = Path(save_dir)
        self.save_dir.mkdir(parents=True, exist_ok=True)
    
    def plot_learning_curve(self, rewards, window=100, filename='learning_curve.png'):
        """绘制学习曲线"""
        # 计算移动平均
        rewards = np.array(rewards)
        moving_avg = np.convolve(rewards, np.ones(window)/window, mode='valid')
        
        plt.figure(figsize=(12, 6))
        
        # 原始奖励
        plt.subplot(2, 1, 1)
        plt.plot(rewards, alpha=0.3, label='Raw Reward')
        plt.plot(moving_avg, label=f'Moving Average ({window})')
        plt.xlabel('Episode')
        plt.ylabel('Reward')
        plt.title('Learning Curve')
        plt.legend()
        plt.grid(True)
        
        # 分布图
        plt.subplot(2, 1, 2)
        plt.hist(rewards, bins=50, alpha=0.7, edgecolor='black')
        plt.xlabel('Reward')
        plt.ylabel('Frequency')
        plt.title('Reward Distribution')
        plt.grid(True)
        
        plt.tight_layout()
        plt.savefig(self.save_dir / filename, dpi=150)
        plt.close()
    
    def plot_loss_curves(self, losses, filename='loss_curves.png'):
        """绘制损失曲线"""
        fig, axes = plt.subplots(2, 2, figsize=(12, 8))
        
        # Policy Loss
        if 'policy_loss' in losses[0]:
            policy_losses = [l['policy_loss'] for l in losses]
            self._plot_single_curve(axes[0, 0], policy_losses, 'Policy Loss')
        
        # Value Loss
        if 'value_loss' in losses[0]:
            value_losses = [l['value_loss'] for l in losses]
            self._plot_single_curve(axes[0, 1], value_losses, 'Value Loss')
        
        # Entropy
        if 'entropy' in losses[0]:
            entropies = [l['entropy'] for l in losses]
            self._plot_single_curve(axes[1, 0], entropies, 'Entropy')
        
        # Gradient Norm
        if 'grad_norm' in losses[0]:
            grad_norms = [l['grad_norm'] for l in losses]
            self._plot_single_curve(axes[1, 1], grad_norms, 'Gradient Norm')
        
        plt.tight_layout()
        plt.savefig(self.save_dir / filename, dpi=150)
        plt.close()
    
    def _plot_single_curve(self, ax, data, title, window=100):
        """绘制单条曲线"""
        data = np.array(data)
        moving_avg = np.convolve(data, np.ones(window)/window, mode='valid')
        
        ax.plot(data, alpha=0.3, label='Raw')
        ax.plot(moving_avg, label=f'MA({window})')
        ax.set_title(title)
        ax.set_xlabel('Step')
        ax.legend()
        ax.grid(True)
    
    def plot_episode_length(self, lengths, window=100, filename='episode_length.png'):
        """绘制episode长度"""
        lengths = np.array(lengths)
        moving_avg = np.convolve(lengths, np.ones(window)/window, mode='valid')
        
        plt.figure(figsize=(10, 6))
        plt.plot(lengths, alpha=0.3, label='Raw Length')
        plt.plot(moving_avg, label=f'Moving Average ({window})')
        plt.xlabel('Episode')
        plt.ylabel('Episode Length')
        plt.title('Episode Length Over Time')
        plt.legend()
        plt.grid(True)
        plt.savefig(self.save_dir / filename, dpi=150)
        plt.close()
    
    def create_training_video(self, model, env, filename='trained_agent.mp4', num_episodes=5):
        """创建训练后的agent视频"""
        from matplotlib.animation import FuncAnimation
        import matplotlib.patches as mpatches
        
        fig, ax = plt.subplots(figsize=(8, 8))
        
        frames = []
        for episode in range(num_episodes):
            obs = env.reset()
            done = False
            episode_frames = []
            
            while not done:
                # 渲染环境
                frame = env.render(mode='rgb_array')
                episode_frames.append(frame)
                
                # agent动作
                action, _ = model.predict(obs, deterministic=True)
                obs, reward, done, info = env.step(action)
            
            frames.extend(episode_frames)
        
        # 创建动画
        def animate(frame_idx):
            ax.clear()
            ax.imshow(frames[frame_idx])
            ax.set_title(f'Episode {frame_idx // len(frames[0]) + 1}, Frame {frame_idx % len(frames[0])}')
            return [ax.imshow(frames[frame_idx])]
        
        anim = FuncAnimation(fig, animate, frames=len(frames), interval=50)
        anim.save(self.save_dir / filename, writer='ffmpeg', fps=30)
        plt.close()
```

### 4.1 评估脚本

```python
# scripts/evaluate.py
import argparse
import numpy as np
from stable_baselines3 import PPO, SAC
from stable_baselines3.common.vec_env import VecNormalize
import gymnasium as gym

def evaluate_model(model_path, env_id, num_episodes=100, render=False):
    """评估训练好的模型"""
    # 加载环境
    env = gym.make(env_id, render_mode='human' if render else None)
    
    # 加载归一化统计（如果有）
    vec_normalize_path = model_path.replace('final_model.zip', 'vec_normalize.pkl')
    try:
        env = VecNormalize.load(vec_normalize_path, env)
        env.training = False
        env.norm_reward = False
    except:
        print("No VecNormalize found, using raw environment")
    
    # 加载模型
    model = PPO.load(model_path) if 'PPO' in model_path else SAC.load(model_path)
    
    # 评估
    episode_rewards = []
    episode_lengths = []
    
    for episode in range(num_episodes):
        obs, _ = env.reset()
        done = False
        truncated = False
        episode_reward = 0
        episode_length = 0
        
        while not done and not truncated:
            action, _ = model.predict(obs, deterministic=True)
            obs, reward, done, truncated, info = env.step(action)
            episode_reward += reward
            episode_length += 1
        
        episode_rewards.append(episode_reward)
        episode_lengths.append(episode_length)
        
        if (episode + 1) % 10 == 0:
            print(f"Episode {episode + 1}, Reward: {episode_reward:.2f}")
    
    # 统计结果
    results = {
        'mean_reward': np.mean(episode_rewards),
        'std_reward': np.std(episode_rewards),
        'min_reward': np.min(episode_rewards),
        'max_reward': np.max(episode_rewards),
        'mean_length': np.mean(episode_lengths),
        'std_length': np.std(episode_lengths),
    }
    
    print("\n=== Evaluation Results ===")
    print(f"Mean Reward: {results['mean_reward']:.2f} ± {results['std_reward']:.2f}")
    print(f"Min/Max Reward: {results['min_reward']:.2f} / {results['max_reward']:.2f}")
    print(f"Mean Length: {results['mean_length']:.2f} ± {results['std_length']:.2f}")
    
    env.close()
    
    return results

def main():
    parser = argparse.ArgumentParser(description='Evaluate RL agent')
    parser.add_argument('--model_path', type=str, required=True)
    parser.add_argument('--env_id', type=str, required=True)
    parser.add_argument('--num_episodes', type=int, default=100)
    parser.add_argument('--render', action='store_true')
    args = parser.parse_args()
    
    evaluate_model(args.model_path, args.env_id, args.num_episodes, args.render)

if __name__ == "__main__":
    main()
```

```python
# scripts/batch_evaluate.py
import yaml
from pathlib import Path
import pandas as pd

def batch_evaluate(config_file):
    """批量评估多个模型"""
    with open(config_file, 'r') as f:
        config = yaml.safe_load(f)
    
    results = []
    
    for model_config in config['models']:
        print(f"\nEvaluating {model_config['name']}...")
        
        result = evaluate_model(
            model_path=model_config['path'],
            env_id=model_config['env_id'],
            num_episodes=config.get('num_episodes', 100)
        )
        
        result['model_name'] = model_config['name']
        results.append(result)
    
    # 保存结果
    df = pd.DataFrame(results)
    output_path = Path(config['output_file'])
    df.to_csv(output_path, index=False)
    print(f"\nResults saved to {output_path}")
    
    return df
```

```python
import itertools
from pathlib import Path
import yaml

def grid_search(base_config, param_grid):
    """网格搜索"""
    # 生成参数组合
    param_names = list(param_grid.keys())
    param_values = list(param_grid.values())
    all_combinations = list(itertools.product(*param_values))
    
    print(f"Total combinations to try: {len(all_combinations)}")
    
    results = []
    
    for idx, combination in enumerate(all_combinations):
        # 创建配置
        config = base_config.copy()
        for name, value in zip(param_names, combination):
            config[name] = value
        
        experiment_name = f"search_{idx}"
        config['experiment_name'] = experiment_name
        
        print(f"\n{'='*50}")
        print(f"Running experiment {idx + 1}/{len(all_combinations)}")
        print(f"Parameters: {dict(zip(param_names, combination))}")
        print(f"{'='*50}\n")
        
        # 训练
        try:
            model = train(config)
```

```python
import json
from datetime import datetime
from pathlib import Path

class ExperimentTracker:
    """实验追踪器"""
    
    def __init__(self, project_dir):
        self.project_dir = Path(project_dir)
        self.experiments_dir = self.project_dir / "experiments"
        self.experiments_dir.mkdir(exist_ok=True)
    
    def create_experiment(self, config):
        """创建新实验"""
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        exp_name = f"{config['experiment_name']}_{timestamp}"
        exp_dir = self.experiments_dir / exp_name
        exp_dir.mkdir(exist_ok=True)
        
        # 保存配置
        with open(exp_dir / "config.yaml", 'w') as f:
            yaml.dump(config, f)
        
        # 创建日志文件
        log_file = exp_dir / "training_log.json"
        
        return {
            'exp_dir': exp_dir,
            'log_file': log_file,
            'exp_name': exp_name
        }
    
    def log_training_step(self, exp_info, metrics):
        """记录训练步骤"""
        # 追加到日志文件
        with open(exp_info['log_file'], 'a') as f:
            f.write(json.dumps(metrics) + '\n')
    
    def finish_experiment(self, exp_info, final_results):
        """结束实验"""
        # 保存最终结果
        results_file = exp_info['exp_dir'] / "results.json"
        with open(results_file, 'w') as f:
            json.dump(final_results, f, indent=2)
        
        # 创建实验摘要
        self.create_summary(exp_info['exp_dir'])
    
    def create_summary(self, exp_dir):
        """创建实验摘要"""
        # 读取日志
        log_file = exp_dir / "training_log.json"
        logs = []
        with open(log_file, 'r') as f:
            for line in f:
                logs.append(json.loads(line))
        
        # 创建摘要
        summary = {
            'total_steps': logs[-1]['step'] if logs else 0,
            'final_reward': logs[-1]['reward'] if logs else 0,
            'best_reward': max([log['reward'] for log in logs]) if logs else 0,
            'total_time': logs[-1]['time'] if logs else 0,
        }
        
        # 保存摘要
        with open(exp_dir / "summary.json", 'w') as f:
            json.dump(summary, f, indent=2)
```

```python
from fastapi import FastAPI, HTTPException
import torch
import numpy as np
from pydantic import BaseModel

app = FastAPI(title="RL Model API")

# 加载模型
model = torch.jit.load("model_scripted.pt")
model.eval()

class PredictionRequest(BaseModel):
    observation: list

class PredictionResponse(BaseModel):
    action: int
    action_prob: list

@app.post("/predict", response_model=PredictionResponse)
async def predict(request: PredictionRequest):
    """预测接口"""
    try:
        # 转换输入
        obs = torch.tensor(request.observation, dtype=torch.float32).unsqueeze(0)
        
        # 预测
        with torch.no_grad():
            action_logits = model(obs)
            action_probs = torch.softmax(action_logits, dim=-1)
            action = torch.argmax(action_probs, dim=-1).item()
        
        return PredictionResponse(
            action=action,
            action_prob=action_probs.squeeze(0).tolist()
        )
    
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)
```
