# LESR 提示词模板工具集：自动化框架与实用工具

## 一、提示词模板生成器

### 1.1 通用提示词模板类

```python
import pandas as pd
import numpy as np
from typing import Dict, List, Tuple, Optional
from dataclasses import dataclass
from enum import Enum

class TaskType(Enum):
    """任务类型枚举"""
    MOTION = "motion"           # 运动任务 (HalfCheetah, Hopper)
    NAVIGATION = "navigation"   # 导航任务 (AntMaze, PointMaze)
    MANIPULATION = "manipulation"  # 操作任务 (Fetch, HandManipulate)

class PerformanceMetric(Enum):
    """性能指标类型"""
    CUMULATIVE_REWARD = "cumulative_reward"
    SUCCESS_RATE = "success_rate"
    EFFICIENCY = "efficiency"

@dataclass
class StateDimension:
    """状态维度描述"""
    index: int
    description: str
    min_val: float
    max_val: float
    name: str
    type: str  # 'slide' or 'hinge'
    unit: str

@dataclass
class TaskConfig:
    """任务配置"""
    env_name: str
    task_type: TaskType
    state_dimensions: List[StateDimension]
    task_description: str
    performance_metric: PerformanceMetric
    additional_prompt: str = ""
    key_dimensions: List[int] = None
    target_state_dim: int = None

    def __post_init__(self):
        if self.key_dimensions is None:
            self.key_dimensions = []
        if self.target_state_dim is None:
            self.target_state_dim = len(self.state_dimensions)

class PromptTemplateGenerator:
    """
    提示词模板生成器

    功能:
    1. 从 Excel 配置文件加载任务信息
    2. 生成初始化 Prompt
    3. 生成 COT 反馈 Prompt
    4. 生成迭代改进 Prompt
    5. 自动检测任务类型并添加适当的领域提示
    """

    def __init__(self, observation_path: str):
        """
        初始化生成器

        Args:
            observation_path: Excel 配置文件路径
        """
        self.observation_path = observation_path
        self.obs_file = pd.read_excel(observation_path, header=None)
        self.task_configs = {}

    def load_task_config(self, env_name: str) -> TaskConfig:
        """
        从 Excel 加载任务配置

        Args:
            env_name: 环境名称 (如 'halfcheetah', 'antmaze')

        Returns:
            TaskConfig: 任务配置对象
        """
        # 提取环境名称的基础部分 (小写)
        env_key = env_name.split('-')[0].lower()

        # 读取对应 sheet
        try:
            sheet_data = pd.read_excel(
                self.observation_path,
                header=None,
                sheet_name=env_key
            )
        except ValueError:
            raise ValueError(f"Environment '{env_name}' not found in Excel file")

        # 解析状态维度
        state_dimensions = []
        for idx in range(len(sheet_data)):
            row = sheet_data.iloc[idx]
            if pd.isna(row[1]):  # 跳过空行
                continue

            state_dim = StateDimension(
                index=int(row[0]),
                description=str(row[1]),
                min_val=row[2] if not pd.isna(row[2]) else float('-inf'),
                max_val=row[3] if not pd.isna(row[3]) else float('inf'),
                name=str(row[4]) if not pd.isna(row[4]) else '',
                type=str(row[5]) if not pd.isna(row[5]) else '',
                unit=str(row[6]) if not pd.isna(row[6]) else ''
            )
            state_dimensions.append(state_dim)

        # 提取任务描述
        task_description = str(sheet_data.iloc[0, -1])

        # 自动检测任务类型
        task_type = self._detect_task_type(env_name, task_description)

        # 自动生成领域提示
        additional_prompt = self._generate_additional_prompt(task_type, env_name)

        # 自动检测性能指标
        performance_metric = self._detect_performance_metric(task_type)

        # 识别关键维度
        key_dimensions = self._identify_key_dimensions(
            state_dimensions, task_type
        )

        # 创建任务配置
        config = TaskConfig(
            env_name=env_name,
            task_type=task_type,
            state_dimensions=state_dimensions,
            task_description=task_description,
            performance_metric=performance_metric,
            additional_prompt=additional_prompt,
            key_dimensions=key_dimensions,
            target_state_dim=len(state_dimensions)
        )

        self.task_configs[env_name] = config
        return config

    def _detect_task_type(self, env_name: str, task_description: str) -> TaskType:
        """
        自动检测任务类型

        Args:
            env_name: 环境名称
            task_description: 任务描述

        Returns:
            TaskType: 检测到的任务类型
        """
        env_lower = env_name.lower()

        # 导航任务特征
        if any(keyword in env_lower for keyword in ['maze', 'point', 'fetch']):
            return TaskType.NAVIGATION

        # 操作任务特征
        if any(keyword in env_lower for keyword in ['hand', 'door', 'pen', 'hammer']):
            return TaskType.MANIPULATION

        # 默认为运动任务
        return TaskType.MOTION

    def _generate_additional_prompt(
        self,
        task_type: TaskType,
        env_name: str
    ) -> str:
        """
        自动生成领域提示

        Args:
            task_type: 任务类型
            env_name: 环境名称

        Returns:
            str: 领域提示文本
        """
        env_lower = env_name.lower()

        # 导航/操作任务需要分阶段学习提示
        if task_type in [TaskType.NAVIGATION, TaskType.MANIPULATION]:
            return """
Most Importantly: As for this task, the agent should firstly learn how to coordinate
various parts of the body itself before it finally reach the goal. Like a baby should
learn how to walk before finally walking to the goal. Therefore, when you design state
representation and reward, you should cosider how to make the agent learn to coordinate
various parts of the body as well as how to finally reach the goal.
"""

        # 运动任务通常不需要额外提示
        return ""

    def _detect_performance_metric(self, task_type: TaskType) -> PerformanceMetric:
        """
        检测性能指标类型

        Args:
            task_type: 任务类型

        Returns:
            PerformanceMetric: 性能指标类型
        """
        if task_type in [TaskType.NAVIGATION, TaskType.MANIPULATION]:
            return PerformanceMetric.SUCCESS_RATE
        else:
            return PerformanceMetric.CUMULATIVE_REWARD

    def _identify_key_dimensions(
        self,
        state_dimensions: List[StateDimension],
        task_type: TaskType
    ) -> List[int]:
        """
        识别关键维度

        Args:
            state_dimensions: 状态维度列表
            task_type: 任务类型

        Returns:
            List[int]: 关键维度索引列表
        """
        key_dims = []

        for dim in state_dimensions:
            desc_lower = dim.description.lower()

            # 运动任务：速度相关
            if task_type == TaskType.MOTION:
                if 'velocity' in desc_lower and 'x' in desc_lower:
                    key_dims.append(dim.index)

            # 导航任务：位置和高度相关
            elif task_type == TaskType.NAVIGATION:
                if any(keyword in desc_lower for keyword in ['x-coordinate', 'y-coordinate', 'z-coordinate', 'height']):
                    key_dims.append(dim.index)

            # 操作任务：位置和姿态相关
            elif task_type == TaskType.MANIPULATION:
                if any(keyword in desc_lower for keyword in ['position', 'orientation', 'gripper']):
                    key_dims.append(dim.index)

        return key_dims

    def generate_init_prompt(self, config: TaskConfig) -> str:
        """
        生成初始化 Prompt

        Args:
            config: 任务配置

        Returns:
            str: 初始化 Prompt 模板
        """
        # 构建状态空间详细描述
        detail_content = ""
        for dim in config.state_dimensions:
            detail_content += f"- `s[{dim.index}]`: {dim.description} , the unit is {dim.unit}.\n"

        # 性能指标描述
        if config.performance_metric == PerformanceMetric.SUCCESS_RATE:
            perf_desc = "Final Policy Success Rate"
        else:
            perf_desc = "Final Policy Performance (accumulated reward)"

        # 组装 Prompt
        prompt = f"""
Revise the state representation for a reinforcement learning agent.
=========================================================
The agent's task description is:
{config.task_description}
=========================================================

The current state is represented by a {config.target_state_dim}-dimensional Python NumPy array, denoted as `s`.

Details of each dimension in the state `s` are as follows:
{detail_content}

You should design a task-related state representation based on the the source {config.target_state_dim} dim
to better for reinforcement training, using the detailed information mentioned above to do some caculations,
and feel free to do complex caculations, and then concat them to the source state.

Besides, we want you to design an intrinsic reward function based on the revise_state python function.

That is to say, we will:
1. use your revise_state python function to get an updated state: updated_s = revise_state(s)
2. use your intrinsic reward function to get an intrinsic reward for the task: r = intrinsic_reward(updated_s)
3. to better design the intrinsic_reward, we recommond you use some source dim in the updated_s,
   which is between updated_s[0] and updated_s[{config.target_state_dim - 1}]
4. however, you must use the extra dim in your given revise_state python function,

    # Your state revision implementation goes here

    # Your intrinsic reward code implementation goes here
    return intrinsic_reward
```
"""
        return prompt.strip()

    def generate_cot_prompt(
        self,
        config: TaskConfig,
        training_results: List[Dict]
    ) -> str:
        """
        生成 COT 反馈 Prompt

        Args:
            config: 任务配置
            training_results: 训练结果列表

        Returns:
            str: COT 反馈 Prompt
        """
        # 性能指标描述
        if config.performance_metric == PerformanceMetric.SUCCESS_RATE:
            perf_desc = "Final Policy Success Rate"
        else:
            perf_desc = "Final Policy Performance"

        # 构建训练结果反馈
        s_feedback = ""
        for idx, result in enumerate(training_results):
            s_feedback += f'========== State Revise and Intrinsic Reward Code -- {idx + 1} ==========\n'
            s_feedback += result['code'] + '\n'
            s_feedback += f'========== State Revise and Intrinsic Reward Code -- {idx + 1}\\'s {perf_desc}: {round(result["score"], 2)} ==========\n'

            # Lipschitz 分析
            s_feedback += f'In this State Revise Code {idx + 1}, the source state dim is from s[0] to s[{config.target_state_dim - 1}], the Lipchitz constant between them and the reward are(Note: The Lipschitz constant is always greater than or equal to 0, and a lower Lipschitz constant implies better smoothness.):\n'

            for dim_idx in range(config.target_state_dim):
                if 'lipschitz_source' in result:
                    s_feedback += f's[{dim_idx}] lipschitz constant with reward = {round(result["lipschitz_source"][dim_idx], 2)}\n'

            s_feedback += '\n'

            # 添加维度分析
            if result['added_dim_count'] > 0:
                s_feedback += f'In this State Revise Code {idx + 1}, you give {result["added_dim_count"]} extra dim from s[{config.target_state_dim}] to s[{config.target_state_dim + result["added_dim_count"] - 1}], the lipschitz constant between them and the reward are:\n'

                for added_idx in range(result['added_dim_count']):
                    dim_idx = config.target_state_dim + added_idx
                    if 'lipschitz_added' in result:
                        s_feedback += f's[{dim_idx}] lipschitz constant with reward = {round(result["lipschitz_added"][added_idx], 2)}\n'

            s_feedback += '\n======================================================================\n\n'

        # 组装 COT Prompt
        cot_prompt = f"""
We have successfully trained Reinforcement Learning (RL) policy using {len(training_results)} different
state revision codes and intrinsic reward function codes sampled by you, and each pair of code
is associated with the training of a policy relatively.

Throughout every state revision code's training process, we monitored:
1. The final policy performance({perf_desc}).
2. Most importantly, every state revise dim's Lipschitz constant with the reward. That is to say,
   you can see which state revise dim is more related to the reward and which dim can contribute
   to enhancing the continuity of the reward function mapping. Lower Lipchitz constant means
   better continuity and smoother of the mapping. Note: Lower Lipchitz constant is better.

Here are the results:
{s_feedback}

You should analyze the results mentioned above and give suggestions about how to imporve the
performace of the "state revision code".

Here are some tips for how to analyze the results:
(a) if you find a state revision code's performance is very low, then you should analyze to figure out why it fail
(b) if you find some dims' are more related to the final performance, you should analyze to figure out what makes it success
(c) you should also analyze how to imporve the performace of the "state revision code" and "intrinsic reward code" later

Lets think step by step. Your solution should aim to improve the overall performance of the RL policy.
"""
        return cot_prompt.strip()

### 1.2 代码验证工具

```python
import importlib.util
import os
import tempfile
import numpy as np
from typing import Tuple, Optional

class CodeValidator:
    """
    代码验证工具

    功能:
    1. 提取 Python 代码
    2. 验证代码可执行性
    3. 检查维度约束
    4. 检查数值稳定性
    5. 检查输出范围
    """

    @staticmethod
    def extract_python_code(llm_output: str) -> str:
        """
        从 LLM 输出中提取 Python 代码

        Args:
            llm_output: LLM 原始输出

        Returns:
            str: 提取的 Python 代码
        """
        output = llm_output.strip()

        # 格式 1: Markdown 代码块 ```python
        if "```python" in output:
            start = output.find("```python") + 9
            end = output.find("```", start)
            if end != -1:
                return output[start:end].strip()

        # 格式 2: 简单代码块 ```
        if "```" in output:
            start = output.find("```") + 3
            end = output.find("```", start)
            if end != -1:
                return output[start:end].strip()

        # 格式 3: 函数定义标记
        if "def revise_state" in output:
            start = output.find("def revise_state")
            # 找到最后一个 return
            last_return = output.rfind("return")
            if last_return != -1:
                end = last_return + 100  # 包含 return 语句
                return output[start:end].strip()

        # 格式 4: import 语句标记
        if "import numpy" in output:
            start = output.find("import")
            end = len(output)
            return output[start:end].strip()

        # 如果都不匹配，返回原输出
        return output

    @staticmethod
    def validate_code(
        code: str,
        test_state: np.ndarray,
        original_dim: int,
        verbose: bool = True
    ) -> Tuple[bool, Optional[str], Optional[int]]:
        """
        验证生成的代码

        Args:
            code: Python 代码
            test_state: 测试状态
            original_dim: 原始状态维度
            verbose: 是否打印详细信息

        Returns:
            Tuple[bool, Optional[str], Optional[int]]:
                (是否通过, 错误信息, 扩展后的维度)
        """
        try:
            # 1. 保存到临时文件
            with tempfile.NamedTemporaryFile(mode='w', suffix='.py', delete=False) as f:
                f.write(code)
                temp_path = f.name

            # 2. 动态导入
            spec = importlib.util.spec_from_file_location("temp_module", temp_path)
            module = importlib.util.module_from_spec(spec)
            spec.loader.exec_module(module)

            # 3. 测试 revise_state
            revised_state = module.revise_state(test_state)

            # 检查输出类型
            if not isinstance(revised_state, np.ndarray):
                if verbose:
                    print(f"❌ revise_state 输出类型错误: {type(revised_state)}")
                os.remove(temp_path)
                return False, "Output type error", None

            # 检查输出维度
            if revised_state.ndim != 1:
                if verbose:
                    print(f"❌ revise_state 输出维度错误: {revised_state.shape}")
                os.remove(temp_path)
                return False, "Output dimension error", None

            # 检查是否添加了维度
            if len(revised_state) <= original_dim:
                if verbose:
                    print(f"❌ 未添加额外维度: {len(revised_state)} <= {original_dim}")
                os.remove(temp_path)
                return False, "No dimensions added", None

            revised_dim = len(revised_state)

            # 检查数值稳定性
            if np.any(np.isnan(revised_state)):
                if verbose:
                    print(f"❌ revised_state 包含 NaN")
                os.remove(temp_path)
                return False, "NaN detected", None

            if np.any(np.isinf(revised_state)):
                if verbose:
                    print(f"❌ revised_state 包含 Inf")
                os.remove(temp_path)
                return False, "Inf detected", None

            # 4. 测试 intrinsic_reward
            intrinsic_r = module.intrinsic_reward(revised_state)

            # 检查输出类型
            if not isinstance(intrinsic_r, (int, float, np.number)):
                if verbose:
                    print(f"❌ intrinsic_reward 输出类型错误: {type(intrinsic_r)}")
                os.remove(temp_path)
                return False, "Reward type error", None

            # 检查输出范围
            if intrinsic_r < -100 or intrinsic_r > 100:
                if verbose:
                    print(f"❌ intrinsic_reward 超出范围 [-100, 100]: {intrinsic_r}")
                os.remove(temp_path)
                return False, "Reward out of range", None

            # 检查数值稳定性
            if np.isnan(intrinsic_r) or np.isinf(intrinsic_r):
                if verbose:
                    print(f"❌ intrinsic_reward 数值不稳定: {intrinsic_r}")
                os.remove(temp_path)
                return False, "Reward numerical instability", None

            # 所有检查通过
            if verbose:
                print(f"✅ 代码验证通过: {original_dim} → {revised_dim} 维")

            os.remove(temp_path)
            return True, None, revised_dim

        except Exception as e:
            if verbose:
                print(f"❌ 代码验证失败: {e}")
            if 'temp_path' in locals() and os.path.exists(temp_path):
                os.remove(temp_path)
            return False, str(e), None

    @staticmethod
    def batch_validate(
        codes: List[str],
        test_state: np.ndarray,
        original_dim: int,
        verbose: bool = True
    ) -> List[Tuple[bool, Optional[str], Optional[int]]]:
        """
        批量验证代码

        Args:
            codes: 代码列表
            test_state: 测试状态
            original_dim: 原始状态维度
            verbose: 是否打印详细信息

        Returns:
            List[Tuple[bool, Optional[str], Optional[int]]]: 验证结果列表
        """
        results = []
        for idx, code in enumerate(codes):
            if verbose:
                print(f"\n验证样本 {idx + 1}/{len(codes)}...")
            result = CodeValidator.validate_code(
                code, test_state, original_dim, verbose
            )
            results.append(result)
        return results

import numpy as np

def revise_state(s):
    forward_velocity = s[8]
    energy = np.sum(np.abs(s[0:9] * s[9:17]))
    body_height = s[0]

from typing import Dict, List
import numpy as np

class PromptEvaluator:
    """
    提示词有效性评估工具

    功能:
    1. 语义完整性检查
    2. LLM 输出质量检查
    3. 代码质量检查
    4. 任务性能提升检查
    5. 综合评分
    """

    @staticmethod
    def check_semantic_completeness(
        config: TaskConfig,
        prompt: str
    ) -> Dict[str, float]:
        """
        检查语义完整性

        Args:
            config: 任务配置
            prompt: 提示词

        Returns:
            Dict[str, float]: 语义完整性指标
        """
        checks = {}

        # 1. 状态空间覆盖率
        covered_dims = sum(1 for dim in config.state_dimensions if f"s[{dim.index}]" in prompt)
        state_coverage = covered_dims / len(config.state_dimensions)
        checks['state_coverage'] = state_coverage

        # 2. 任务描述长度
        task_desc_len = len(config.task_description)
        checks['task_description_length'] = min(task_desc_len / 1000, 1.0)

        # 3. 单位信息覆盖率
        unit_covered = sum(1 for dim in config.state_dimensions if dim.unit in prompt)
        unit_coverage = unit_covered / len(config.state_dimensions)
        checks['unit_coverage'] = unit_coverage

        # 4. 约束明确性
        required_keywords = [
            "dimensional",
            "[-100, 100]",
            "must use the extra dim"
        ]
        constraint_score = sum(1 for kw in required_keywords if kw in prompt) / len(required_keywords)
        checks['constraint_clarity'] = constraint_score

        return checks

    @staticmethod
    def check_llm_output_quality(
        generated_codes: List[str],
        validation_results: List[Tuple[bool, str, int]]
    ) -> Dict[str, float]:
        """
        检查 LLM 输出质量

        Args:
            generated_codes: 生成的代码列表
            validation_results: 验证结果列表

        Returns:
            Dict[str, float]: LLM 输出质量指标
        """
        checks = {}

        if not validation_results:
            return checks

        # 1. 格式正确率
        format_correct = sum(1 for r in validation_results if r[0] or "syntax" not in r[1].lower())
        format_correct_rate = format_correct / len(validation_results)
        checks['format_correct_rate'] = format_correct_rate

        # 2. 约束满足率
        constraint_satisfied = sum(1 for r in validation_results if r[0])
        constraint_satisfied_rate = constraint_satisfied / len(validation_results)
        checks['constraint_satisfied_rate'] = constraint_satisfied_rate

        # 3. 多样性 (代码编辑距离)
        if len(generated_codes) >= 2:
            diversity_scores = []
            for i in range(len(generated_codes) - 1):
                for j in range(i + 1, len(generated_codes)):
                    dist = PromptEvaluator._code_edit_distance(
                        generated_codes[i], generated_codes[j]
                    )
                    max_len = max(len(generated_codes[i]), len(generated_codes[j]))
                    diversity = dist / max_len if max_len > 0 else 0
                    diversity_scores.append(diversity)

            avg_diversity = np.mean(diversity_scores) if diversity_scores else 0
            checks['diversity'] = avg_diversity
        else:
            checks['diversity'] = 0.0

        return checks

    @staticmethod
    def _code_edit_distance(code1: str, code2: str) -> int:
        """
        计算代码编辑距离 (简化版)

        Args:
            code1: 代码1
            code2: 代码2

        Returns:
            int: 编辑距离
        """
        # 简化版：使用行数差异
        lines1 = set(code1.split('\n'))
        lines2 = set(code2.split('\n'))
        return len(lines1.symmetric_difference(lines2))

    @staticmethod
    def check_code_quality(
        validation_results: List[Tuple[bool, str, int]],
        lipschitz_data: List[np.ndarray] = None
    ) -> Dict[str, float]:
        """
        检查代码质量

        Args:
            validation_results: 验证结果列表
            lipschitz_data: Lipschitz 常数数据

        Returns:
            Dict[str, float]: 代码质量指标
        """
        checks = {}

        if not validation_results:
            return checks

        # 1. 验证通过率
        validation_pass = sum(1 for r in validation_results if r[0])
        validation_pass_rate = validation_pass / len(validation_results)
        checks['validation_pass_rate'] = validation_pass_rate

        # 2. 特征有效性 (基于 Lipschitz)
        if lipschitz_data:
            effectiveness_scores = []
            for lipschitz in lipschitz_data:
                if len(lipschitz) > 27:  # 有添加维度
                    added_dims = lipschitz[27:]
                    effective = np.sum(added_dims > 0.01) / len(added_dims)
                    effectiveness_scores.append(effective)

            avg_effectiveness = np.mean(effectiveness_scores) if effectiveness_scores else 0
            checks['feature_effectiveness'] = avg_effectiveness
        else:
            checks['feature_effectiveness'] = 0.0

        # 3. 平均扩展维度
        revised_dims = [r[2] for r in validation_results if r[2] is not None]
        if revised_dims:
            avg_revised_dim = np.mean(revised_dims)
            checks['avg_revised_dim'] = avg_revised_dim
        else:
            checks['avg_revised_dim'] = 0.0

        return checks

    @staticmethod
    def check_task_performance(
        iteration_scores: List[float],
        baseline_score: float
    ) -> Dict[str, float]:
        """
        检查任务性能提升

        Args:
            iteration_scores: 每轮迭代最佳分数
            baseline_score: 基线分数

        Returns:
            Dict[str, float]: 任务性能指标
        """
        checks = {}

        if not iteration_scores:
            return checks

        # 1. 最终性能提升率
        best_score = max(iteration_scores)
        if baseline_score > 0:
            improvement_rate = (best_score - baseline_score) / baseline_score
        else:
            improvement_rate = 0.0
        checks['improvement_rate'] = improvement_rate

        # 2. 收敛迭代数
        target_score = baseline_score + 0.9 * (best_score - baseline_score)
        convergence_iter = 0
        for i, score in enumerate(iteration_scores):
            if score >= target_score:
                convergence_iter = i + 1
                break
        checks['convergence_iter'] = convergence_iter if convergence_iter > 0 else len(iteration_scores)

        # 3. 平均每轮提升
        if len(iteration_scores) > 1:
            avg_improvement_per_iter = np.mean(np.diff(iteration_scores))
            checks['avg_improvement_per_iter'] = avg_improvement_per_iter
        else:
            checks['avg_improvement_per_iter'] = 0.0

        # 4. 性能方差
        score_variance = np.var(iteration_scores)
        checks['score_variance'] = score_variance

        return checks

    @staticmethod
    def calculate_overall_score(
        semantic_checks: Dict[str, float],
        llm_output_checks: Dict[str, float],
        code_quality_checks: Dict[str, float],
        task_performance_checks: Dict[str, float]
    ) -> Dict[str, any]:
        """
        计算综合评分

        Args:
            semantic_checks: 语义完整性检查
            llm_output_checks: LLM 输出质量检查
            code_quality_checks: 代码质量检查
            task_performance_checks: 任务性能检查

        Returns:
            Dict[str, any]: 综合评分结果
        """
        # 权重
        weights = {
            'semantic': 0.15,
            'llm_output': 0.25,
            'code_quality': 0.30,
            'task_performance': 0.30
        }

        # Level 1: 语义完整性
        semantic_score = (
            semantic_checks.get('state_coverage', 0) * 0.4 +
            semantic_checks.get('task_description_length', 0) * 0.3 +
            semantic_checks.get('unit_coverage', 0) * 0.3
        )

        # Level 2: LLM 输出质量
        llm_output_score = (
            llm_output_checks.get('format_correct_rate', 0) * 0.3 +
            llm_output_checks.get('constraint_satisfied_rate', 0) * 0.4 +
            min(llm_output_checks.get('diversity', 0), 1.0) * 0.3
        )

        # Level 3: 代码质量
        code_quality_score = (
            code_quality_checks.get('validation_pass_rate', 0) * 0.3 +
            code_quality_checks.get('feature_effectiveness', 0) * 0.4 +
            min(code_quality_checks.get('avg_revised_dim', 0) / 50, 1.0) * 0.3
        )

        # Level 4: 任务性能
        task_performance_score = (
            min(task_performance_checks.get('improvement_rate', 0) * 2, 1.0) * 0.4 +
            (1 - task_performance_checks.get('convergence_iter', 10) / 10) * 0.2 +
            min(task_performance_checks.get('avg_improvement_per_iter', 0) / 1000, 1.0) * 0.2 +
            max(1 - task_performance_checks.get('score_variance', 50000) / 50000, 0) * 0.2
        )

        # 综合评分
        overall_score = (
            semantic_score * weights['semantic'] +
            llm_output_score * weights['llm_output'] +
            code_quality_score * weights['code_quality'] +
            task_performance_score * weights['task_performance']
        )

        # 评级
        if overall_score >= 90:
            grade = "A+ (优秀)"
        elif overall_score >= 80:
            grade = "A (良好)"
        elif overall_score >= 70:
            grade = "B (中等)"
        elif overall_score >= 60:
            grade = "C (及格)"
        else:
            grade = "D (需要改进)"

        return {
            'overall_score': overall_score * 100,
            'grade': grade,
            'breakdown': {
                'semantic': semantic_score * 100,
                'llm_output': llm_output_score * 100,
                'code_quality': code_quality_score * 100,
                'task_performance': task_performance_score * 100
            },
            'weights': weights
        }

class LESRPromptToolkit:
    """
    LESR 提示词工具集

    整合所有工具的端到端框架
    """

    def __init__(self, observation_path: str):
        """
        初始化工具集

        Args:
            observation_path: Excel 配置文件路径
        """
        self.generator = PromptTemplateGenerator(observation_path)
        self.validator = CodeValidator()
        self.evaluator = PromptEvaluator()

    def create_prompt_for_task(
        self,
        env_name: str,
        verbose: bool = True
    ) -> Tuple[TaskConfig, str]:
        """
        为任务创建提示词

        Args:
            env_name: 环境名称
            verbose: 是否打印详细信息

        Returns:
            Tuple[TaskConfig, str]: (任务配置, 初始化提示词)

            print(f"  任务类型: {config.task_type.value}")
            print(f"  状态维度: {len(config.state_dimensions)}")
            print(f"  关键维度: {config.key_dimensions}")
            print(f"  性能指标: {config.performance_metric.value}")
            if config.additional_prompt:
                print(f"  领域提示: 已添加")
            else:
                print(f"  领域提示: 不需要")

        # 生成提示词
        init_prompt = self.generator.generate_init_prompt(config)

        if verbose:
            print(f"  ✅ 提示词生成完成")

        return config, init_prompt

    def validate_generated_codes(
        self,
        codes: List[str],
        env_name: str,
        verbose: bool = True
    ) -> List[Tuple[bool, str, int]]:
        """
        验证生成的代码

        Args:
            codes: 代码列表
            env_name: 环境名称
            verbose: 是否打印详细信息

        Returns:
            List[Tuple[bool, str, int]]: 验证结果列表
        """
        if env_name not in self.configs:
            raise ValueError(f"Environment '{env_name}' not configured")

        test_state = np.random.rand(len(config.state_dimensions))

        if verbose:
            print(f"\n验证 {len(codes)} 个代码样本...")

        results = self.validator.batch_validate(
            codes, test_state, len(config.state_dimensions), verbose
        )

        if verbose:
            valid_count = sum(1 for r in results if r[0])
            print(f"\n验证通过率: {valid_count}/{len(codes)} ({valid_count/len(codes)*100:.1f}%)")

        return results

    def evaluate_prompt_effectiveness(
        self,
        env_name: str,
        prompt: str,
        generated_codes: List[str],
        validation_results: List[Tuple[bool, str, int]],
        iteration_scores: List[float],
        baseline_score: float,
        verbose: bool = True

        """

        Args:

            generated_codes: 生成的代码
            validation_results: 验证结果
            iteration_scores: 迭代分数

        Returns:
            Dict[str, any]: 评估结果

        config = self.configs[env_name]

        if verbose:

        )

        # 5. 综合评分

            semantic_checks,
            llm_output_checks,
            code_quality_checks,
            task_performance_checks
